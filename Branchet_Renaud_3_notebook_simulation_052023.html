ari_score = []

    df_periode= df_synt[(df_synt['order_delivered_customer_date'].dt.date- 
                         df_synt['order_delivered_customer_date'].dt.date.min()).dt.days//30.5<3+p]
    df_periode= df_periode.groupby("customer_unique_id").agg({'recence': 'min', 'order_id': "nunique", 'payment_value' : 'sum', 
                                                              'Satisfaction_num' : 'mean',
                                                              "freight_value": "sum",
                                                              "nb_payment_sequential": "mean", 
                                                              "sum_payment_installments": "mean", 
                                                              "delta_order_delivered_customer_date": "mean",
                                                              "sale_month": lambda x:x.value_counts().index[0], 
                                                              'distance_client_vendeur': 'mean'})

    # Rename columns
    df_periode = df_periode.rename(columns={'order_id':'frequence','payment_value' : 'montant',
                                            "freight_value": "total_freight",
                                            "nb_payment_sequential": "mean_payment_sequential",
                                            "sum_payment_installments": "mean_payment_installments",
                                            "delta_order_delivered_customer_date": "mean_delivery_days",
                                            "sale_month": "favorite_sale_month"})
    #préprocessing   
    # Transformation logarithmique des variables 
    for i in ['mean_delivery_days', 'montant','distance_client_vendeur','total_freight']:
        df_periode[i] = np.log(df_periode[i]+1)

    
    # Filter on initial customer client
    df_periode = df_periode[df_periode.index.isin(df_init.index)]
    init_labels = kmeans_init.predict(df_periode)
    # K-Means
    kmeans_p = Pipeline([("preprocessor", RobustScaler()),
                         ("kmeans", KMeans(n_clusters=4, n_init=10, random_state=42))])
    kmeans_p.fit(df_periode)
    p_labels = kmeans_p.named_steps['kmeans'].labels_
    
    # Calculate best k
    sse = []
    for i in range(2, 8):
        kmeans = KMeans(n_clusters=i, n_init=10, random_state=42)
        kmeans.fit(df_periode)
        sse.append(kmeans.inertia_)
    kl = KneeLocator(range(2, 8), sse, curve="convex", direction="decreasing")
    bestk.append(kl.elbow)
    
    # Calculate ARI score

    ari_score.append([p+3,ari_p])
    silhouette_scores.append(silhouette_score(df_periode, kmeans_init.fit_predict(df_periode)))

Conclusion et perspectives¶

Nous avons segmenté notre dataset clients en utilisant les algorithmes K-Means et DBSCAN, les resultats tournent autour de 3 à 5 clusters, mais le plus probant est le modele K-Means qui donne des clusters bien définis et distincts. Par ailleurs, La réduction de dimension ne perturbe pas la segmentation, permet de réduire le nombre de variables mais dégrade légèrement le silhouette-score. Avec notre modèle, l' analyse en composantes principales n'est pas indispensable, mais il faudrait s'y résoudre si on complexifiait le modèle avec un ajout conséquent de variables. Je propose quelques perspectives d'approfondissement du modèle:

  • Analyse textuelle des commentaires clients pour mieux évaluer la satisfaction.
  • Ajouter des variables: catégorie de produits achetés.
  • Ajouter les valeurs extrêmes au modèle ou les traiter séparément

Par ailleurs, l' étude de l'évolution du clustering dans le temps montre une chute des scores ari et silhouette autour du 10e mois. Il sera donc nécessaire de réévaluer le clustering tous les 11 mois. On pourrait aussi proposer de nouveaux modèles en cas de changements de stratégies marketing de l'entreprise ou durant les périodes de ventes exceptionnelles( black friday, fêtes de fin d'année, soldes)
